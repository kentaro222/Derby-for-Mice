function setupRaceParams() { mouses.forEach(m => { const baseSpeed = 0.2 + Math.random() * 0.5; const isTripper = baseSpeed > 0.5 && Math.random() < 0.6; const tripPercent = 0.4 + Math.random() * 0.4; const isLateCharger = baseSpeed < 0.4 && Math.random() < 0.6; // ★変更箇所(案A): スパートの発動位置を 40%〜65% の間に前倒し // (元は 0.5 + Math.random() * 0.3) const chargePercent = 0.4 + Math.random() * 0.25; m.speedProfile = { base: baseSpeed, fluctuation: 0.5 + Math.random() * 1.0, isTripper: isTripper, tripPercent: tripPercent, hasTripped: false, isLateCharger: isLateCharger, chargePercent: chargePercent, hasCharged: false, isSuperAwakeningCandidate: false, isSuperAwakening: false }; }); // ★変更箇所(案C): 超覚醒の発生確率を 5% から 15% に大幅アップ！ // (元は 0.05) if (Math.random() < 0.15) { let chosenId = Math.floor(Math.random() * 6); mouses[chosenId].speedProfile.isSuperAwakeningCandidate = true; } }
